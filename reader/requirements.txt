#   pip install -r requirements.txt

# Offline install (all wheels are pre-bundled in ../wheels/):
#   pip install --no-index --find-links=../wheels -r requirements.txt

# NOTE — PyTorch / CUDA:
#   Do NOT list torch/torchaudio here. setup.py installs the correct CUDA/CPU
#   build first. Including them in this file causes `pip install -r` to pull
#   a CPU wheel from PyPI and overwrite the CUDA install (breaks TTS load).
#   Manual GPU install if needed:
#     pip install torch torchaudio --index-url https://download.pytorch.org/whl/cu128

# Tell pip to also search the bundled wheels when doing an offline install.
--find-links=../wheels
